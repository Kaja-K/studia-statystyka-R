Funkcje identyfikujące obserwacje odstające

odstajace3 <- function(y){
   q1_q3 <- quantile(y, probs =c(0.25, 0.75))
   myIQR <- IQR(y)
   which(y > q1_q3[2] + 3*myIQR |  y < q1_q3[1] - 3*myIQR )
}
odstajace1 <- function(y){
   q1_q3 <- quantile(y, probs =c(0.25, 0.75))
   myIQR <- IQR(y)
   which(y > q1_q3[2] + 1.5*myIQR |  y < q1_q3[1] - 1.5*myIQR )
}

Usuń obserwacje odstające

nowyX <- data.frame()
for(i in modele){
   temp <- filter(x, model==i)
   kto <- odstajace1(temp$cenaPLN) 
   nowyX <- rbind(nowyX, temp[-kto, ])
}

Możemy szybciej usunąć obserwacje odstające, bez używania pętli. Trochę zmienimy funkcję odstajace

odstajace <- function(y){
  q1_q3 <- quantile(y, probs =c(0.25, 0.75))
  myIQR <- IQR(y)
  (y <= q1_q3[2] + 1.5*myIQR) &  (y >= q1_q3[1] - 1.5*myIQR)
}
nowyX <- x %>% group_by(model) %>% filter(odstajace(cenaPLN))
